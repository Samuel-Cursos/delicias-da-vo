SUBSTITUA AS REGRAS INTEIRAS POR ESTAS:

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function logado() {
      return request.auth != null;
    }

    function isAdmin() {
      return logado()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data.tipo == "admin";
    }

    match /usuarios/{uid} {
      allow create: if logado() && request.auth.uid == uid;
      allow read: if logado() && (request.auth.uid == uid || isAdmin());
      allow update, delete: if isAdmin();
    }

    match /produtos/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /categorias/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /promocoes/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /configuracoes/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /salgadosFesta/{produtoId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }

    match /contadoresPedidos/{data} {
      allow read: if isAdmin();
      allow create, update: if true;
      allow delete: if isAdmin();
    }

    match /pedidosSite/{data} {
      allow read: if isAdmin();

      match /pedidos/{pedidoId} {
        allow create: if true;
        allow read, update, delete: if isAdmin();
      }
    }

    // Necessário para a Central e o Backup consultarem pedidos de todas as datas
    match /{caminho=**}/pedidos/{pedidoId} {
      allow read: if isAdmin();
    }

    match /vendas/{id} {
      allow read, write: if isAdmin();
    }

    // Contador público usado apenas para gerar DV-0001, DV-0002...
    match /contadores/encomendasFesta {
      allow read, create, update: if true;
      allow delete: if isAdmin();
    }

    // Outros contadores são exclusivos do administrador
    match /contadores/{id} {
      allow read, write: if isAdmin();
    }

    match /encomendasFesta/{pedidoId} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }

    match /movimentosFinanceiros/{movimentoId} {
      allow read, create, update, delete: if isAdmin();
    }

    match /custosProdutos/{produtoId} {
      allow read, create, update, delete: if isAdmin();
    }

    match /fechamentosFinanceiros/{mesId} {
      allow read, create, update, delete: if isAdmin();
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
